# 03 - 数据访问层设计（Repository Pattern）

## 🎯 目标

引入 Repository Pattern，分离数据查询和业务逻辑，提升代码可测试性和可维护性。

## 📊 当前问题

### Server Actions 混合职责

```typescript
// lib/actions/posts.ts - 当前实现

  // ❌ 问题 1：直接在 action 中写查询逻辑

    }

  if (!post) {
    throw new Error('Post not found')
  }

  // ❌ 问题 2：数据转换逻辑分散

    ...post,

  }
}
```

**问题**：
1. 查询逻辑和业务逻辑耦合
2. 数据转换分散在各处
3. 难以复用查询
4. 难以单元测试
5. Server Actions 代码臃肿（平均 152 行）

### 步骤 2：实现 PostRepository

```typescript
// lib/repositories/post.repository.ts - 新建

import { db } from '@/lib/db'
import { posts, postsToTags } from '@/lib/db/schema'
import { eq, and, desc, ilike, count } from 'drizzle-orm'
import { BaseRepository } from './base.repository'
import { NotFoundError } from '@/lib/utils/error'
import type {
  Post,
  PostWithRelations,
  PostSummary,
  AdminPostListItem,
  PaginatedResult,
} from '@/types/models'
import {
  transformPostWithRelations,
  transformPostSummary,
} from '@/types/transformers'

/**
 * 文章数据访问层
 */
export class PostRepository extends BaseRepository<Post> {
  protected tableName = 'posts'

  /**
   * 根据 slug 查询文章（带完整关系）
   */
  async findBySlug(slug: string): Promise<PostWithRelations> {
    const post = await db.query.posts.findFirst({
      where: eq(posts.slug, slug),
      with: {
        author: {
          columns: {
            id: true,
            username: true,
            name: true,
            avatar: true,
            bio: true,
          },
        },
        category: true,
        tags: {
          with: {
            tag: true,
          },
        },
      },
    })

    if (!post) {
      throw new NotFoundError('文章未找到')
    }

    return transformPostWithRelations(post)
  }

  /**
   * 根据 ID 查询文章（带完整关系）
   */
  async findByIdWithRelations(id: string): Promise<PostWithRelations> {
    const post = await db.query.posts.findFirst({
      where: eq(posts.id, id),
      with: {
        author: {
          columns: {
            id: true,
            username: true,
            name: true,
            avatar: true,
            bio: true,
          },
        },
        category: true,
        tags: {
          with: {
            tag: true,
          },
        },
      },
    })

    if (!post) {
      throw new NotFoundError('文章未找到')
    }

    return transformPostWithRelations(post)
  }

  /**
   * 查询已发布文章列表（分页）
   */
  async findPublished(params: {
    page?: number
    pageSize?: number
    category?: string
    search?: string
  }): Promise<PaginatedResult<PostSummary>> {
    const { page = 1, pageSize = 10, category, search } = params
    const offset = (page - 1) * pageSize

    // 构建查询条件
    const conditions = [eq(posts.published, true)]
    if (search) {
      conditions.push(ilike(posts.title, `%${search}%`))
    }
    if (category) {
      conditions.push(eq(posts.categoryId, category))
    }

    // 查询数据
    const postsList = await db.query.posts.findMany({
      where: and(...conditions),
      with: {
        author: {
          columns: {
            id: true,
            username: true,
            name: true,
            avatar: true,
            bio: true,
          },
        },
        category: true,
        tags: {
          with: {
            tag: true,
          },
        },
      },
      orderBy: desc(posts.createdAt),
      limit: pageSize,
      offset,
    })

    // 查询总数
    const [totalResult] = await db
      .select({ count: count() })
      .from(posts)
      .where(and(...conditions))

    const total = totalResult.count

    return {
      data: postsList.map(transformPostSummary),
      total,
      page,
      pageSize,
      totalPages: Math.ceil(total / pageSize),
      hasMore: page * pageSize < total,
    }
  }

  /**
   * 查询作者的文章列表（管理后台）
   */
  async findByAuthor(authorId: string): Promise<AdminPostListItem[]> {
    const postsList = await db.query.posts.findMany({
      where: eq(posts.authorId, authorId),
      with: {
        category: true,
      },
      orderBy: desc(posts.createdAt),
    })

    return postsList.map((post) => ({
      id: post.id,
      title: post.title,
      slug: post.slug,
      published: post.published,
      views: post.views,
      likesCount: post.likesCount,
      commentsCount: post.commentsCount,
      createdAt: post.createdAt,
      category: {
        id: post.category.id,
        name: post.category.name,
        slug: post.category.slug,
        description: post.category.description,
      },
    }))
  }

  /**
   * 创建文章
   */
  async create(data: {
    title: string
    slug: string
    content: string
    excerpt?: string
    coverImage?: string
    published: boolean
    authorId: string
    categoryId: string
    tagIds: string[]
  }): Promise<Post> {
    const { tagIds, ...postData } = data

    // 创建文章
    const [post] = await db
      .insert(posts)
      .values(postData)
      .returning()

    // 关联标签
    if (tagIds.length > 0) {
      await db.insert(postsToTags).values(
        tagIds.map((tagId) => ({
          postId: post.id,
          tagId,
        }))
      )
    }

    return post
  }

  /**
   * 更新文章
   */
  async update(
    id: string,
    data: Partial<{
      title: string
      slug: string
      content: string
      excerpt: string
      coverImage: string
      published: boolean
      categoryId: string
      tagIds: string[]
    }>
  ): Promise<Post> {
    const { tagIds, ...postData } = data

    // 更新文章
    const [post] = await db
      .update(posts)
      .set(postData)
      .where(eq(posts.id, id))
      .returning()

    // 更新标签关联
    if (tagIds !== undefined) {
      // 删除旧关联
      await db.delete(postsToTags).where(eq(postsToTags.postId, id))

      // 创建新关联
      if (tagIds.length > 0) {
        await db.insert(postsToTags).values(
          tagIds.map((tagId) => ({
            postId: id,
            tagId,
          }))
        )
      }
    }

    return post
  }

  /**
   * 删除文章
   */
  async delete(id: string): Promise<void> {
    await db.delete(posts).where(eq(posts.id, id))
  }

  /**
   * 增加浏览量
   */
  async incrementViews(id: string): Promise<void> {
    await db
      .update(posts)
      .set({ views: db.raw('views + 1') })
      .where(eq(posts.id, id))
  }

  /**
   * 检查 slug 是否存在
   */
  async slugExists(slug: string, excludeId?: string): Promise<boolean> {
    const conditions = [eq(posts.slug, slug)]
    if (excludeId) {
      conditions.push(ne(posts.id, excludeId))
    }

    const post = await db.query.posts.findFirst({
      where: and(...conditions),
      columns: { id: true },
    })

    return !!post
  }

  /**
   * 转换数据库结果
   */
  protected transform(raw: any): Post {
    return raw as Post
  }
}
```

### 步骤 3：实现 CommentRepository

```typescript
// lib/repositories/comment.repository.ts - 新建

import { db } from '@/lib/db'
import { comments } from '@/lib/db/schema'
import { eq, and, isNull, desc } from 'drizzle-orm'
import { BaseRepository } from './base.repository'
import type { Comment, CommentWithAuthor } from '@/types/models'
import { transformComment } from '@/types/transformers'

/**
 * 评论数据访问层
 */
export class CommentRepository extends BaseRepository<Comment> {
  protected tableName = 'comments'

  /**
   * 查询文章的所有评论（含嵌套回复）
   */
  async findByPostId(postId: string): Promise<CommentWithAuthor[]> {
    // 查询顶级评论
    const topLevelComments = await db.query.comments.findMany({
      where: and(
        eq(comments.postId, postId),
        isNull(comments.parentId)
      ),
      with: {
        author: {
          columns: {
            id: true,
            username: true,
            name: true,
            avatar: true,
            bio: true,
          },
        },
      },
      orderBy: desc(comments.createdAt),
    })

    // 查询所有回复
    const replies = await db.query.comments.findMany({
      where: and(
        eq(comments.postId, postId),
        isNotNull(comments.parentId)
      ),
      with: {
        author: {
          columns: {
            id: true,
            username: true,
            name: true,
            avatar: true,
            bio: true,
          },
        },
      },
      orderBy: desc(comments.createdAt),
    })

    // 构建嵌套结构
    const repliesMap = new Map<string, CommentWithAuthor[]>()
    replies.forEach((reply) => {
      const parentId = reply.parentId!
      if (!repliesMap.has(parentId)) {
        repliesMap.set(parentId, [])
      }
      repliesMap.get(parentId)!.push(transformComment(reply))
    })

    // 组装结果
    return topLevelComments.map((comment) => ({
      ...transformComment(comment),
      replies: repliesMap.get(comment.id) || [],
    }))
  }

  /**
   * 创建评论
   */
  async create(data: {
    content: string
    authorId: string
    postId: string
    parentId?: string
  }): Promise<Comment> {
    const [comment] = await db
      .insert(comments)
      .values(data)
      .returning()

    return comment
  }

  /**
   * 删除评论
   */
  async delete(id: string): Promise<void> {
    await db.delete(comments).where(eq(comments.id, id))
  }

  /**
   * 转换数据库结果
   */
  protected transform(raw: any): Comment {
    return raw as Comment
  }
}
```

```typescript
// lib/actions/posts.ts - 改造后（简化到 180 行）
'use server'

import { revalidatePath } from 'next/cache'
import { postRepo } from '@/lib/repositories'
import { requireAuth, requireResourceAccess } from '@/lib/auth/server-permissions'
import { postFormSchema } from '@/types/forms'
import type { PostFormData } from '@/types/forms'

/**
 * 根据 slug 获取文章
 */
export async function getPostBySlug(slug: string) {
  return postRepo.findBySlug(slug)
}

/**
 * 创建文章
 */
export async function createPost(data: PostFormData) {
  const session = await requireAuth()

  // 验证数据
  const validated = postFormSchema.parse(data)

  // 检查 slug 是否存在
  const slugExists = await postRepo.slugExists(validated.slug)
  if (slugExists) {
    throw new Error('该 Slug 已被使用')
  }

  // 创建文章
  const post = await postRepo.create({
    ...validated,
    authorId: session.user.id,
  })

  // 失效缓存
  revalidatePath('/blog')
  revalidatePath('/')

  return post
}

/**
 * 更新文章
 */
export async function updatePost(postId: string, data: PostFormData) {
  await requireAuth()

  // 获取文章
  const existingPost = await postRepo.findByIdWithRelations(postId)

  // 权限检查
  await requireResourceAccess(existingPost.author.id)

  // 验证数据
  const validated = postFormSchema.parse(data)

  // 检查 slug（排除当前文章）
  if (validated.slug !== existingPost.slug) {
    const slugExists = await postRepo.slugExists(validated.slug, postId)
    if (slugExists) {
      throw new Error('该 Slug 已被使用')
    }
  }

  // 更新文章
  const post = await postRepo.update(postId, validated)

  // 失效缓存
  revalidatePath('/blog')
  revalidatePath(`/blog/${validated.slug}`)
  revalidatePath('/')

  return post
}

/**
 * 删除文章
 */
export async function deletePost(postId: string) {
  await requireAuth()

  // 获取文章
  const existingPost = await postRepo.findByIdWithRelations(postId)

  // 权限检查
  await requireResourceAccess(existingPost.author.id)

  // 删除文章
  await postRepo.delete(postId)

  // 失效缓存
  revalidatePath('/blog')
  revalidatePath('/')

  return { success: true }
}

/**
 * 获取已发布文章列表
 */
export async function getPosts(params: {
  page?: number
  limit?: number
  category?: string
  search?: string
}) {
  return postRepo.findPublished({
    page: params.page || 1,
    pageSize: params.limit || 10,
    category: params.category,
    search: params.search,
  })
}
```
